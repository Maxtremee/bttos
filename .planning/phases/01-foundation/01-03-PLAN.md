```tsx
import { MemoryRouter, Route } from '@solidjs/router'
import { createMemoryHistory } from '@solidjs/router'
import { createSignal, Show, onMount, onCleanup } from 'solid-js'
import LoginScreen from './screens/LoginScreen'
import ChannelsScreen from './screens/ChannelsScreen'
import PlayerScreen from './screens/PlayerScreen'
import SettingsScreen from './screens/SettingsScreen'
import ExitConfirmDialog from './components/ExitConfirmDialog'

// Root screens: Back on these triggers exit confirmation (D-03)
const ROOT_PATHS = ['/', '/login', '/channels']

const KEY_BACK = 461 // webOS remote Back button (keyCode 0x1CD)

const history = createMemoryHistory()

export default function App() {
  const [showExitDialog, setShowExitDialog] = createSignal(false)

  function handleBack() {
    const currentPath = history.get().value ?? '/'

    if (showExitDialog()) {
      // Back while dialog is open: dismiss dialog (same as Cancel)
      setShowExitDialog(false)
      return
    }

    if (ROOT_PATHS.includes(currentPath)) {
      // On a root screen: show exit confirmation dialog (D-03)
      setShowExitDialog(true)
    } else {
      // On a non-root screen: navigate back in history
      history.set({ value: currentPath === '/settings' ? '/channels' : '/channels' })
    }
  }

  function handleKeydown(e: KeyboardEvent) {
    if (e.keyCode === KEY_BACK) {
      e.preventDefault()
      handleBack()
    }
  }

  onMount(() => window.addEventListener('keydown', handleKeydown))
  onCleanup(() => window.removeEventListener('keydown', handleKeydown))

  return (
    <>
      <MemoryRouter history={history} root={(props) => <>{props.children}</>}>
        <Route path="/" component={LoginScreen} />
        <Route path="/login" component={LoginScreen} />
        <Route path="/channels" component={ChannelsScreen} />
        <Route path="/player/:channel" component={PlayerScreen} />
        <Route path="/settings" component={SettingsScreen} />
      </MemoryRouter>
      <Show when={showExitDialog()}>
        <ExitConfirmDialog
          onExit={() => window.close()}
          onCancel={() => setShowExitDialog(false)}
        />
      </Show>
    </>
  )
}
```

Key constraints:
- D-01: MemoryRouter is the router — NO BrowserRouter, NO HashRouter
- D-02: No CSS transition classes, no animation on screen swap — MemoryRouter replaces DOM synchronously
- D-03: Back on root screens shows ExitConfirmDialog, not immediate exit
- `KEY_BACK = 461` — webOS Back button keyCode; do NOT use keyCode 8 (browser back/backspace)
- ExitConfirmDialog is rendered OUTSIDE the MemoryRouter so it overlays everything
- The `Show` component from solid-js controls dialog visibility
- Do NOT destructure signal values — always call `showExitDialog()` (it is an Accessor/getter function)
- `history.get().value` returns the current path; `history.get()` may return `{ value: string, state: unknown }`
- The back navigation for non-root screens is simplified here (always to /channels); real back-stack will be addressed in later phases when actual navigation exists

import { onMount } from 'solid-js'
import { FocusableGroup, Focusable, useSpatialNavigation } from '../navigation'

interface ExitConfirmDialogProps {
  onExit: () => void
  onCancel: () => void
}

export default function ExitConfirmDialog(props: ExitConfirmDialogProps) {
  const { setFocus } = useSpatialNavigation()
  onMount(() => setFocus('exit-confirm-exit-btn'))

  return (
    // Full-screen overlay — sits above all screens
    <div style={{
      position: 'fixed',
      inset: '0',
      background: 'rgba(0, 0, 0, 0.75)',
      display: 'flex',
      'align-items': 'center',
      'justify-content': 'center',
      'z-index': '1000',
    }}>
      <FocusableGroup focusKey="exit-confirm" isFocusBoundary={true}>
        <div style={{
          background: 'var(--color-surface)',
          padding: 'var(--space-3xl)',
          display: 'flex',
          'flex-direction': 'column',
          gap: 'var(--space-2xl)',
          'min-width': '400px',
        }}>
          <h2 style={{
            'font-size': 'var(--font-size-heading)',
            'font-weight': 'var(--font-weight-semibold)',
            color: 'var(--color-text-primary)',
            'text-align': 'center',
          }}>
            Exit app?
          </h2>
          <div style={{
            display: 'flex',
            gap: 'var(--space-lg)',
            'justify-content': 'center',
          }}>
            <Focusable focusKey="exit-confirm-exit-btn" as="div">
              {({ focused }) => (
                <button
                  class={focused() ? 'focused' : ''}
                  onClick={() => props.onExit()}
                  style={{
                    'min-height': 'var(--min-target-height)',
                    padding: 'var(--space-md) var(--space-2xl)',

                    'font-weight': 'var(--font-weight-semibold)',

                    color: 'var(--color-text-primary)',
                    border: 'none',
                    cursor: 'pointer',
                  }}
                >
                  Exit
                </button>
              )}
            </Focusable>
            <Focusable focusKey="exit-confirm-cancel-btn" as="div">
              {({ focused }) => (
                <button
