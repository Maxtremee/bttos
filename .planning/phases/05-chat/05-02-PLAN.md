```typescript
const [emoteMap, setEmoteMap] = createSignal<EmoteMap>(new Map())

// Chat message store — capped at 150 messages
const MAX_MESSAGES = 150
const [messages, setMessages] = createStore<ChatMessage[]>([])

// Message batching — buffer for 100ms before flushing to store
let pendingMessages: ChatMessage[] = []
let batchTimer: ReturnType<typeof setTimeout> | undefined

function queueMessage(msg: ChatMessage) {
  pendingMessages.push(msg)
  if (!batchTimer) {
    batchTimer = setTimeout(flushMessages, 100)
  }
}

function flushMessages() {
  if (!pendingMessages.length) return
  setMessages(prev => {
    const next = [...prev, ...pendingMessages]
    pendingMessages = []
    return next.length > MAX_MESSAGES ? next.slice(-MAX_MESSAGES) : next
  })
  batchTimer = undefined
}
```

```typescript
async function initChat(broadcasterId: string) {
  setChatStatus('connecting')

  // Set up chat service callbacks
  twitchChatService.onMessage = queueMessage
  twitchChatService.onScopeError = () => setScopeError(true)
  twitchChatService.onConnectionChange = (connected) => {
    setChatStatus(connected ? 'active' : 'reconnecting')
  }

  // Connect to EventSub
  twitchChatService.connect(broadcasterId, authStore.userId!, authStore.token!)

  // Fetch emote map
  setChatStatus('loading-emotes')
  const map = await emoteService.getEmoteMap(broadcasterId)
  setEmoteMap(map)
  setChatStatus('active')
}

// Start chat when stream metadata resolves (provides broadcaster ID)

  initChat(broadcasterId)
})
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| User input (keydown) -> toggle state | Remote button press changes UI state only — no security impact |
| Router state -> broadcasterId | Value originates from Helix API response (trusted), passed through client-side router state |
| ChatMessage data -> DOM rendering | Message text rendered via SolidJS JSX (auto-escaped), emotes rendered as img src from known CDNs |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-07 | Spoofing | ChatMessage username color | accept | Color is a CSS property from EventSub data; worst case is an ugly color. SolidJS style binding does not allow script injection via CSS values. |
| T-05-08 | Tampering | ChatMessage text content | mitigate | SolidJS JSX auto-escapes text content. No innerHTML or dangerouslySetInnerHTML used. Text fragments rendered as text nodes. |
| T-05-09 | Tampering | Emote img src URLs | mitigate | Twitch emote URLs constructed from known CDN patterns with only the emote ID variable. Third-party emote URLs from EmoteMap are fetched from known provider APIs. No user-controlled URL construction. |
| T-05-10 | Information Disclosure | Router state broadcasterId | accept | Broadcaster ID is public information (visible in Twitch URLs). No sensitive data in router state. |
| T-05-11 | Denial of Service | Rapid toggle spam | accept | Toggle only changes a boolean signal — no network calls, no expensive operations. SolidJS handles reactive updates efficiently. |
</threat_model>
